var a = 1, b = 1, c, d;

c = ++a; alert(c); // 2 --- по префиксному написанию увеличивает а до 2 и присваивает в с значение 2

d = b++; alert(d); // 1 ---постфиксный плюс увеличивает b на 1 и устанавливает d = 1

c = (2+ ++a); alert(c); // 5 ---а уже увеличено и равно 2, увеличиваем еще на 1 и прибавляем 2, ответ 5

d = (2+ b++); alert(d); // 4 --- b на этом шаге 2 и добавляем 2, увеличение b будет после

alert(a); // 3 --- когда выводили в с, там а уже было равно 3

alert(b); // 3 --- после вывода значения d, произошло постфиксное увеличение b и стало равным 3
